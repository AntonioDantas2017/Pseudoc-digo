função numeroPrimo (numero : inteiro) : inteiro
	se (numero > 0) então
		para i de 2 ate numero - 1 faça
			se numero % i então
				retorna 0
			fim-se
		fim-para
		retorna 1
	senão
		retorna -1
	fim-se
fim-função
