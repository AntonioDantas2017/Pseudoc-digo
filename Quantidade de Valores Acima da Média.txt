função TotalAlunos ( V[máximoAlunos] : real ) : inteiro
	declare 
		i : inteiro  {Variável de Controle}
		soma : real  {Soma das Notas}
		contador : inteiro {Soma dos alunos com nota acima da média}

	{Somando todas as Notas}
	para i de 0 até máximoAlunos faça
		soma <- soma + V[i]	
	fim-para

	{Soma dos alunos com nota acima da média}
	para i de 0 até máximoAlunos faça
		se V[i] >= (soma/máximoAlunos) então
			contador <- contador + 1
		fim-se
	fim-para
	
	retorne contador
fim-função
